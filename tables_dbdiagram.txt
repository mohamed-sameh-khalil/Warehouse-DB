Table Entity{
  ID INTEGER [pk]
  name VARCHAR(100)
  phone VARCHAR(20)
  email VARCHAR(25)
  fax VARCHAR(30)
  country VARCHAR(20)
  state VARCHAR(20)
  city VARCHAR(20)
  street VARCHAR(20)
  building VARCHAR(20)
}
Table Supplier{
  ID INTEGER [pk, ref: > Entity.ID]
  contract_start DATE
  contract_end DATE
}

Table Items {
  ID INTEGER [pk]
  arrival_time TIMESTAMP
  color_pallete VARCHAR(10)
  status varchar
  location_letter CHAR(1)
  location_num CHAR(3)
  supplier_ID INTEGER [ref: > Supplier.ID]
  distributor_ID INTEGER [ref: > Distributor.ID]
  Category_ID INTEGER [ref: > Category.ID]
}


Table Distributor{
  ID INTEGER [pk, ref: > Entity.ID]
  tax_No VARCHAR(40)
}

Table Company{
  ID INTEGER [pk, ref: > Distributor.ID]
  register_No VARCHAR(40)
}

Table Person{
  ID INTEGER [pk, ref: > Distributor.ID]
  SSN VARCHAR(20)
}

Table TV {
  ID INTEGER [pk, ref: > Category.ID]
  screensize FLOAT
  Model_No VARCHAR(20)
  screentype VARCHAR(10)
}


Table Chargers {
  ID INTEGER [pk, ref: > Category.ID]
  voltage FLOAT
}


Table Category{
  ID INTEGER [pk]
  name VARCHAR(100)
  weight FLOAT
  company VARCHAR(50)
  length FLOAT
  width FLOAT
  height FLOAT
  desc VARCHAR(500)
}

Table Mobiles {
  ID INTEGER [pk, ref: > Category.ID]
  screensize FLOAT
  Model_No VARCHAR(20)
}

Table Video_Games {
  ID INTEGER [pk, ref: > Category.ID]
  genre VARCHAR(20)
  release_date DATE
}

Table platform{
  name VARCHAR(20) [pk]
  version VARCHAR(20) [pk]
  video_game_id INTEGER [pk, ref: > Video_Games.ID]
}


Table Color{
  color_pallete VARCHAR(10) [pk]
  Category_ID INTEGER [pk, ref: > Category.ID]
}
